\tikzset{
    basiclistnode/.style={
        draw,
        rectangle,
        thick,
        minimum size=20pt,
        text=black,
    }
}

\tikzset{
    listelementcontent/.style={
        basiclistnode,
        fill=blue!15,
    }
}

\tikzset{
    emptynode/.style={
        basiclistnode,
        fill=orange!60,
        path picture = {
            \draw[-, black]
            (path picture bounding box.north west) --
            (path picture bounding box.south east)
            (path picture bounding box.south west) --
            (path picture bounding box.north east);
        }
    }
}

\tikzset{
    sllistnode/.style={
        basiclistnode,
        rectangle split,
        rectangle split parts=2, draw,
        rectangle split horizontal,
        rectangle split part fill={red!20, blue!20}
    }
}

\tikzset{
    pointerbox/.style={
        basiclistnode,
        fill={blue!15}
    }
